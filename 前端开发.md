    if (err) return res.json({ status: 500, msg: '失败' })
    res.json({ status: 200, data: { result } })

})

  console.log('服务器运行在 3001 端口上')

## 七、封装网络请求工具类（核心）

### 在 utils 下新建 request.js

javascript



运行









```
// 封装 wx.request
function request(url, method, data) {
  wx.showLoading({ title: '加载中...' })
  return new Promise((resolve, reject) => {
    wx.request({
      url: url,
      method: method,
      data: data,
      header: {
        'content-type': 'application/x-www-form-urlencoded'
      },
      success(res) {
        resolve(res)
      },
      fail(err) {
        reject(err)
      },
      complete() {
        wx.hideLoading()
      }
    })
  })
}

// 接口基础地址
const baseUrl = 'http://localhost:3001'

// 接口列表
const api = {
  banner: '/api/banner',
  goods: '/api/goods',
  hotSearch: '/api/keywords',
  search: '/api/goods/search',
  goodsDetails: '/api/goods/details',
  cart: '/api/cart',
  addCart: '/api/cart/add',
  delCart: '/api/cart/del',
  category: '/api/category',
  buy: '/api/buy',
  login: '/api/login'
}

// 导出请求方法
export function getBanner(data = {}) {
  return request(baseUrl + api.banner, 'GET', data)
}

export function getGoods(data = {}) {
  return request(baseUrl + api.goods, 'GET', data)
}

export default request
```

## 八、页面编写（index 页面）

### 1. index.js

javascript



运行









```
import { getBanner, getGoods } from '../../utils/request.js'

Page({
  data: {
    bannerList: [],
    goodList: []
  },

  onLoad() {
    this.getBannerData()
    this.getGoodsData()
  },

  // 获取轮播图
  async getBannerData() {
    try {
      const res = await getBanner()
      this.setData({
        bannerList: res.data.data.result
      })
    } catch (err) {
      console.error('轮播图请求失败', err)
    }
  },

  // 获取商品列表
  async getGoodsData() {
    try {
      const res = await getGoods()
      this.setData({
        goodList: res.data.data.result
      })
    } catch (err) {
      console.error('商品列表请求失败', err)
    }
  }
})
```

### 2. index.wxml

xml











```
<!-- 轮播图 -->
<view wx:for="{{bannerList}}" wx:key="index">
  {{index}} —— {{item.url}}
</view>

<!-- 商品列表 -->
<view wx:for="{{goodList}}" wx:key="index">
  {{index}} —— {{item.name}} —— ￥{{item.price}}
</view>
```
