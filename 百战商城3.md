### 触底加载实现

**配置触底距离**：可以在 `app.json` 或页面的 `.json` 文件中通过 `onReachBottomDistance` 属性调整触发触底事件的提前量（默认是 50px）

`pages\index\index.json`   :  设置提前量为 60px

```json
{
  "usingComponents": {
    "banner":"/components/banner/banner"
  },
  "onReachBottomDistance": 60
}
```

**监听触底事件**： `onReachBottom`  监听页面滚动到底部的事件。

`pages\index\index.js` :   

```js
// index.js
//导入request.js
import {getGoods} from "../../utils/request.js"

Page({
  data: {
    page:1,       //当前页
    goodList: [], //商品列表
  },

  onLoad() {
      //获得商品列表第一页
      this.getGoodsWarp(1)
  },
  //触底事件回调  
  onReachBottom() {
    // 更改页码
    this.setData({
      page: this.data.page += 1
    })
    //获得商品列表的下一页
    this.getGoodsWarp(this.data.page)
  },
  //获得商品列表
  getGoodsWarp(page) {
    //发送网络请求获得商品列表
    getGoods({page}).then(res => {
       // res.data.msg === true  表示已经到底了
      if (!res.data.msg) {
        this.setData({
          // 老数据合并新数据，做累加操作
          goodList: this.data.goodList.concat(res.data.data.result)
        })
      } else {
        // 给出用户提示：没有数据了
        wx.showToast({
          title: res.data.msg,
          icon: "success",
          duration: 2000
        })
      }
    })
  }

})
```

`pages\details\details.js`

```javascript
// pages/details/details.js
import {
  getGoodsDetails
} from "../../utils/request.js"

Page({
  data: {
    goodsDetails: {}
  },
  onLoad(options) {
    // 提示用户在获取数据
    wx.showLoading({
      title: '等待数据加载...',
    })
    //获得商品详情页
    getGoodsDetails({
      id: options.id
    }).then(res => {
      wx.hideLoading()
      if (res.data.status === 200) {
        this.setData({
          goodsDetails: res.data.data[0]
        })
      } else {
        wx.showToast({
          title: '数据获取失败',
          icon: "success"
        })
      }
    })
  }
})
```

`pages\user\user.js`

```javascript
import { getLogin } from '../../utils/request'

Page({
    data: {
        userInfo: {},
        list:[
          {
              text:"前端",
              icon:"like",
              color:"#ff0000"
          },
          {
              text:"Python",
              icon:"star",
              color:"#2a83fe"
          },
          {
              text:"Java",
              icon:"fire",
              color:"#fd6012"
          }
      ]
    },
    onLoad(options) {
        if(wx.getStorageSync('userInfo')){
            this.setData({
                userInfo:wx.getStorageSync('userInfo')
            })
        }
    },
    getUserProfile() {
        wx.getUserProfile({
            desc: "展示用户信息",
            success: res => {
                console.log("UserProfile=",res); 
                this.setData({
                    userInfo:res.userInfo
                })
                this.loginHandle()
                wx.setStorageSync('userInfo', res.userInfo)
            },
            fail(err) {
                console.log(err);
            },
            complete() {
                console.log("获取完成");
            }
        })
    },
    loginHandle(){
        wx.login({
            success(response){
                let code = response.code;
                getLogin({code:code}).then(res =>{
                  let   session_key = res.data.data
                  wx.setStorageSync('session_key', session_key)
                })
            },
            fail(err){
                console.log(err);
            }
        })
    }
})
```
